# (爬虫)部门是公司的核心业务部门
# 负责创建requests对象和接受引擎发送过来的response（Downloader部门爬取到的内容），从中解析并提取出有用的数据。
# 它对应的是爬虫流程【解析数据】和【提取数据】这两步
import scrapy
import bs4
import re
# 需要引用DoubanItem，它在items里面。因为是items在top250.py的上一级目录，所以要用..items，这是一个固定用法。
from ..items import DoubanItem
# DoubanSpider类继承自scrapy.Spider类
class DoubanSpider(scrapy.Spider):
    name = 'music'   # 定义爬虫的名字，这个名字是爬虫的唯一标识
    # allowed_domains是定义允许爬虫爬取的网址域名（不需要加https://）。
    # 如果网址的域名不在这个列表里，就会被过滤掉
    # 一定在book.douban.com这个域名之下，防止爬到广告页面去
    allowed_domains = ['music.douban.com']
    # 定义起始网址，就是爬虫从哪个网址开始抓取
    # 注：allowed_domains的设定对start_urls里的网址不会有影响
    start_urls = ['https://music.douban.com/top250']

    # 根据url的规律，构建出前三页地址
    start_urls = []
    for x in range(0,10):
        url = 'https://music.douban.com/top250?start=' + str(x * 25)
        start_urls.append(url)
    
    # 解析下载器传过来的数据
    def parse(self, response):
        if response.status == 200:
            # 爬取的数据存储在response.text
            bs = bs4.BeautifulSoup(response.text,'html.parser')
            datas = bs.find_all('tr',class_="item")
            for data in  datas:
                #实例化DoubanItem类。
                item = DoubanItem()
                # 标题
                item['title'] = data.find_all('a')[1].text
                # 封面
                item['fengmian'] = data.find_all('img')[0]['src']
                # 评分
                item['rate'] = data.find('span',class_='rating_nums').text
                # 评论数
                rate_count = data.find('span',class_='pl').text
                pattern = re.compile('(\d*?)人评价')
                item['rate_count'] = re.findall(pattern,rate_count)[0]
                # 剩下的要进入详情页继续爬
                href = data.find_all('a')[1]['href']
                yield scrapy.Request(href, callback=self.parse2,meta={
                    'title':item['title'],
                    'fengmian':item['fengmian'],
                    'rate':item['rate'],
                    'rate_count':item['rate_count']
                })

    # 解析详情页
    def parse2(self, response):
        bs = bs4.BeautifulSoup(response.text,'html.parser')
        item = DoubanItem()
        item['title'] = response.meta['title']
        item['fengmian'] = response.meta['fengmian']
        item['rate'] = response.meta['rate']
        item['rate_count'] = response.meta['rate_count']
        # 歌手、发行日期、类型、介质、曲风
        pattern = re.compile(r'<a href="/search.*?">(.*?)</a>')
        item['geshou'] = re.findall(pattern,response.text)[0]
        pattern = re.compile(r'\d{4}-\d{2}-\d{2}')
        item['date'] = re.findall(pattern,response.text)[0]
        pattern = re.compile(r'<span class="pl">专辑类型:</span>&nbsp;(.*?)\n')
        item['type'] = re.findall(pattern,response.text)[0]
        pattern = re.compile(r'<span class="pl">介质:</span>&nbsp;(.*?)\n')
        item['jiezhi'] = re.findall(pattern,response.text)[0]
        pattern = re.compile(r'<span class="pl">流派:</span>&nbsp;(.*?)\n')
        item['qufeng'] = re.findall(pattern,response.text)[0]
        yield item

    if SnowNLP(i).sentiments>=0.6:
        qinggan = '正向'

        qinggan = '中性'

对情感倾向分数小于等于0.4的评论信息标注为负向

import pandas as pd
from snownlp import SnowNLP

df = pd.read_csv('movie_comment_all.csv')
def snow(i):
    if SnowNLP(i).sentiments>=0.6:
        qinggan = '正向'
    elif 0.6>SnowNLP(i).sentiments>0.4:
        qinggan = '中性'
    else:
        qinggan = '负向'
    return qinggan

df['评分'] = df['content'].apply(lambda x:SnowNLP(x).sentiments)
df['情感倾向'] = df['content'].map(snow)
print(df[["评分", "情感倾向",'content']].sample(5)) # 随机抽取n行

df.to_csv('standard.csv',encoding='utf-8')
